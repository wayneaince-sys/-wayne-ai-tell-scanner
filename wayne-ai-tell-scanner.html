    function tokenizeSentences(text) {

      return text.replace(/\n+/g, ' ').match(/[^.!?]+[.!?]+|[^.!?]+$/g)?.map(s => s.trim()).filter(Boolean) || [];

    function tokenizeWords(text) {

    function average(arr) {

      return arr.length ? arr.reduce((a, b) => a + b, 0) / arr.length : 0;

    }


    function stdDev(arr) {

      if (!arr.length) return 0;

      const mean = average(arr);

      const variance = average(arr.map(n => (n - mean) ** 2));

      return Math.sqrt(variance);

    }

    function generateSentenceRewrite(sentence) {

      let revised = sentence.trim();

      revised = revised.replace(/\b(In today's world|Overall|Furthermore|Additionally|However|In conclusion|It is important to note that|It is worth noting that)\b[,:]?\s*/gi, '');

      revised = revised.replace(/\b(arguably|perhaps|may well|seems to|it appears that|in many ways)\b\s*/gi, '');

      revised = revised.replace(/\b(multifaceted|holistic|robust|meaningful|complex)\b/gi, match => ({multifaceted:'layered', holistic:'whole', robust:'strong', meaningful:'real', complex:'hard'}[match.toLowerCase()] || match));

      revised = revised.replace(/\bframework\b/gi, 'structure');

      revised = revised.replace(/\bconversation\b/gi, 'fight');

      revised = revised.replace(/\bstakeholders\b/gi, 'players');

      revised = revised.replace(/\s{2,}/g, ' ').trim();

      if (revised && !/[.!?]$/.test(revised)) revised += '.';

      if (revised.length && revised[0] === revised[0].toLowerCase()) revised = revised[0].toUpperCase() + revised.slice(1);

      return revised || sentence;

    }

        });

        if (hits > 0) {

          const points = Math.min(group.weight * hits, group.weight * 3);

          score += points;

          findings.push({ label: group.label, hits, points, detail: `${hits} trigger${hits > 1 ? 's' : ''} caught in this category.` });

          scoreBreakdown.push({ label: group.label, points });

        }

      });


      const repeatedWords = Object.entries(wordFreq).filter(([word, count]) => count >= 5 && word.length > 5).sort((a, b) => b[1] - a[1]).slice(0, 6);

      if (repeatedWords.length) {

        const points = Math.min(14, repeatedWords.length * 3);

        score += points;

        findings.push({ label: 'Repetition drag', hits: repeatedWords.length, points, detail: `Repeated terms: ${repeatedWords.map(([word, count]) => `${word} (${count})`).join(', ')}.` });

        scoreBreakdown.push({ label: 'Repetition', points });

      }


      const sentenceLengths = sentences.map(s => tokenizeWords(s).length);

      const spread = stdDev(sentenceLengths);

      const meanLen = average(sentenceLengths);

      let cadencePoints = 0;

      if (sentences.length >= 4 && spread < 4.5) cadencePoints += 12;

      if (meanLen > 28) cadencePoints += 10;

      if (sentences.length >= 5 && sentences.filter(len => len < 8).length === 0) cadencePoints += 6;

      if (cadencePoints > 0) {

        score += cadencePoints;

        findings.push({ label: 'Flat cadence', hits: cadencePoints, points: cadencePoints, detail: `Average sentence length ${meanLen.toFixed(1)} words; variation score ${spread.toFixed(1)}.` });

        scoreBreakdown.push({ label: 'Cadence', points: cadencePoints });

      }


      const passiveHints = (text.match(/\b(is|was|were|be|been|being)\s+\w+ed\b/gi) || []).length;

      if (passiveHints >= 2) {

        const points = Math.min(10, passiveHints * 2);

        score += points;

        findings.push({ label: 'Passive drift', hits: passiveHints, points, detail: `${passiveHints} passive-looking construction${passiveHints > 1 ? 's' : ''} spotted.` });

        scoreBreakdown.push({ label: 'Passive', points });

      }


      const topSentences = sentences

        .map(sentence => {

          let sentenceScore = 0;

          fillerPatterns.forEach(group => {

            group.patterns.forEach(pattern => {

              const matches = sentence.match(pattern);

              if (matches) sentenceScore += matches.length * group.weight;

            });

          });

          if (tokenizeWords(sentence).length > 28) sentenceScore += 5;

          if (/\b(is|was|were|be|been|being)\s+\w+ed\b/i.test(sentence)) sentenceScore += 4;

          return { sentence, sentenceScore, rewrite: generateSentenceRewrite(sentence) };

        })

        .filter(item => item.sentenceScore > 0)

        .sort((a, b) => b.sentenceScore - a.sentenceScore)

        .slice(0, 3);


      score = Math.min(100, Math.round(score));

      const band = score >= 70 ? 'High risk' : score >= 40 ? 'Moderate risk' : score > 0 ? 'Low risk' : 'Clean read';

      const top = findings.sort((a, b) => b.points - a.points)[0] || null;

      const rewrites = findings.slice(0, 5).map(item => ({ label: item.label, prompt: rewriteTemplates[item.label] || 'Tighten the line. Use sharper nouns, live verbs, and less cushion.' }));


      const markdown = [

        '# Wayne AI Tell Scanner Report',

        '',

        `- Risk band: **${band}**`,

        `- AI-sheen score: **${score}/100**`,

        `- Word count: **${words.length}**`,

        `- Sentence count: **${sentences.length}**`,

        `- Cadence spread: **${spread.toFixed(1)}**`,

        '',

        '## Score breakdown',

        ...(scoreBreakdown.length ? scoreBreakdown.map(item => `- ${item.label}: ${item.points}`) : ['- No scored categories triggered.']),

        '',

        '## Top findings',

        ...(findings.length ? findings.map(item => `- **${item.label}** — ${item.detail}`) : ['- No obvious AI-sheen patterns detected.']),

        '',

        '## Rewrite marching orders',

        ...(rewrites.length ? rewrites.map(item => `- **${item.label}** — ${item.prompt}`) : ['- Keep the pressure. Keep the specificity. Read it aloud once before publishing.']),

        '',

        '## Side-by-side rewrites',

        ...(topSentences.length ? topSentences.flatMap((item, index) => [`### Line ${index + 1}`, `- Original: ${item.sentence}`, `- Rewrite: ${item.rewrite}`, '']) : ['- No rewrite targets pulled in this pass.'])

      ].join('\n');


      const report = [

        'WAYNE AI TELL SCANNER REPORT',

        '--------------------------------',

        `Risk band: ${band}`,

        `AI-sheen score: ${score}/100`,

        `Word count: ${words.length}`,

        `Sentence count: ${sentences.length}`,

        `Cadence spread: ${spread.toFixed(1)}`,

        '',

        'Score breakdown:',

        ...(scoreBreakdown.length ? scoreBreakdown.map(item => `- ${item.label}: ${item.points}`) : ['- No scored categories triggered.']),

        '',

        'Top findings:',

        ...(findings.length ? findings.map(item => `- ${item.label}: ${item.detail}`) : ['- No obvious AI-sheen patterns detected.']),

        '',

        'Rewrite marching orders:',

        ...(rewrites.length ? rewrites.map(item => `- ${item.label}: ${item.prompt}`) : ['- Keep the pressure. Keep the specificity. Read it aloud once before publishing.']),

        '',

        'Side-by-side rewrites:',

        ...(topSentences.length ? topSentences.flatMap((item, index) => [`Line ${index + 1} original: ${item.sentence}`, `Line ${index + 1} rewrite: ${item.rewrite}`, '']) : ['- No rewrite targets pulled in this pass.'])

      ].join('\n');


      return { score, band, top, findings, rewrites, report, markdown, wordTotal: words.length, sentenceTotal: sentences.length, spread: spread.toFixed(1), scoreBreakdown, topSentences };

    }


    function renderResults(result) {

      latestResult = result;

      scoreValue.textContent = result.score;

      riskBand.textContent = result.band;

      topIssue.textContent = result.top ? result.top.label : 'No dominant issue';

      summaryLine.textContent = result.findings.length ? `${result.findings.length} pattern groups triggered. Start with ${result.top.label.toLowerCase()}.` : 'No major AI-sheen patterns triggered in this pass.';

      meterBar.style.width = `${result.score}%`;

      sentenceCount.textContent = result.sentenceTotal;

      wordCount.textContent = result.wordTotal;

      varianceScore.textContent = result.spread;

      flagCount.textContent = result.findings.length;


      scoreGrid.innerHTML = result.scoreBreakdown.length ? result.scoreBreakdown.map(item => `<div class="meta-card"><div class="stat-label">${item.label}</div><div class="stat-value">${item.points}</div><div>points</div></div>`).join('') : '<div class="meta-card"><div class="stat-label">Score grid</div><div>Nothing tripped in this pass.</div></div>';


      flagList.innerHTML = result.findings.length ? result.findings.map(item => `<article class="flag"><strong>${item.label}</strong><div>${item.detail}</div><small>Penalty: ${item.points} points</small></article>`).join('') : '<article class="flag"><strong>Clean read</strong><div>No major AI-sheen patterns jumped off the page in this pass.</div><small>Read it aloud anyway. That is where weak joints get exposed.</small></article>';


      rewriteList.innerHTML = result.rewrites.length ? result.rewrites.map(item => `<div class="rewrite-item"><strong>${item.label}</strong><div>${item.prompt}</div></div>`).join('') : '<div class="rewrite-item"><strong>Holding line</strong><div>The draft reads clean. The next pass should push argument pressure, not AI-sheen cleanup.</div></div>';


      originalCompare.innerHTML = result.topSentences.length ? result.topSentences.map((item, index) => `<div class="compare-line"><strong>Line ${index + 1}</strong><div>${item.sentence}</div></div>`).join('') : '<div class="compare-line"><strong>No hot line found</strong><div>No sentence crossed the rewrite threshold in this pass.</div></div>';

      rewriteCompare.innerHTML = result.topSentences.length ? result.topSentences.map((item, index) => `<div class="compare-line"><strong>Line ${index + 1}</strong><div>${item.rewrite}</div></div>`).join('') : '<div class="compare-line"><strong>No rewrite needed</strong><div>The draft did not cough up a top rewrite target.</div></div>';


      reportBox.textContent = result.report;

    }

      await ensureLibrary();

      }

    }


    function downloadBlob(content, filename, type) {

      const blob = new Blob([content], { type });

      const url = URL.createObjectURL(blob);

      const a = document.createElement('a');

      a.href = url;

      a.download = filename;

      a.click();

      URL.revokeObjectURL(url);

    }
